#Pins to activate on BB black on Linux terminal

#pwm :
cd /sys/devices/bone_capemgr.*
echo am33xx_pwm > slots
echo bone_pwm_P9_14 > slots
cd /sys/devices/ocp.3/pwm_test_P9_14.*
echo 0 > run
echo 0 > polarity
sleep 0.1
cd /sys/devices/bone_capemgr.*
echo bone_pwm_P9_31 > slots 
cd /sys/devices/ocp.3/pwm_test_P9_31.*
echo 0 > run
echo 0 > polarity
sleep 0.1
cd /sys/devices/bone_capemgr.*
echo bone_pwm_P8_13 > slots
cd /sys/devices/ocp.3/pwm_test_P8_13.*
echo 0 > run
echo 0 > polarity
sleep 0.1

#gpio :
cd /sys/class/gpio
echo 30 > export
echo 31 > export
echo 47 > export
echo 27 > export
echo 67 > export
echo 68 > export 
cd /sys/class/gpio/gpio31
echo 0 > value

#ain : 
cd /sys/devices/bone_capemgr.*
echo cape-bone-iio > slots

#eqep 
echo bone_eqep0 > /sys/devices/bone_capemgr.9/slots
echo bone_eqep1 > /sys/devices/bone_capemgr.9/slots
echo bone_eqep2 > /sys/devices/bone_capemgr.9/slots
